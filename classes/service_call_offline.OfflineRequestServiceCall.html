constructor

  • new OfflineRequestServiceCall(options: OfflineRequestOptions & IServiceCallParameters): OfflineRequestServiceCall

  • Overrides ServiceCall.constructor

    • Defined insrc/service-call-offline/index.ts:66

    Creates an OfflineRequestServiceCall helper class If there's no network connection, saves the request to perform later when network connection is available

    example
    import { OfflineRequestServiceCall } from '@smartface/extension-utils/lib/service-call-offline';
    sc = new OfflineRequestServiceCall({
        baseUrl: "http://smartface.io",
        logEnabled: true,
        offlineRequestHandler: requestOptions => {
            return new Promise((resolve, reject) => {
                amce.createRequestOptions(amceOptions)
                    .then(({ headers }) => {
                        resolve(Object.assign({}, requestOptions, headers));
                    });
            });
        }
    });
